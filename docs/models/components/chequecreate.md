# ChequeCreate

## Example Usage

```typescript
import { ChequeCreate } from "postgrid-node/models/components";

let value: ChequeCreate = {
  to: {
    addressLine1: "891 Makenzie Route",
    countryCode: "PK",
    skipVerification: false,
    forceVerifiedStatus: false,
    companyName: "McGlynn - Schiller",
  },
  bankAccount: "<value>",
  amount: 796818,
  from: "<value>",
};
```
