# BoxCreate

## Example Usage

```typescript
import { BoxCreate } from "postgrid-node/models/components";

let value: BoxCreate = {
  to: "<value>",
  from: "<value>",
  cheques: [
    {
      bankAccount: "<value>",
      amount: 100226,
      number: 919483,
      from: "<value>",
      to: "<value>",
    },
  ],
};
```
