# Postcard

## Example Usage

```typescript
import { Postcard } from "postgrid-node/models/components";

let value: Postcard = {
  status: "printing",
  id: "<id>",
  live: false,
  createdAt: new Date("2023-02-13T10:02:48.141Z"),
  updatedAt: new Date("2023-03-07T16:46:29.356Z"),
  sendDate: new Date("2023-03-08T22:36:35.420Z"),
  mailingClass: "royal_mail_first_class",
  to: {
    id: "<id>",
    live: false,
    createdAt: new Date("2025-03-21T21:54:23.669Z"),
    updatedAt: new Date("2025-09-11T06:37:56.079Z"),
    object: "contact",
    addressLine1: "539 Kovacek Highway",
    countryCode: "WS",
    addressStatus: "verified",
  },
  object: "postcard",
  size: "6x4",
};
```
